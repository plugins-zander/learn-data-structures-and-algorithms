#include<iostream>
using namespace std;
int find(int a[],int n){
	if(n<=0)
		return 0;
	int ret=0;
	for(int i=1;i<n;i++)
		if(a[0]>a[i])
			ret++;
	ret+=find(a+1,n-1);
	return ret;
}
